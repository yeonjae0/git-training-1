1.버블정렬을 위한 슈도 코드나 파이썬 코드 완성

a = [55, 7, 78, 12, 42]
N = 5
def bubble_sort(a, N):
    for j in range(N):
        for i in range(N - 1):
            if a[i] > a[i+1]:
                a[i], a[i+1] = a[i+1], a[i]
    return a
